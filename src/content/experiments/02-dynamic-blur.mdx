<div class="w-full" slot="html">

<div class="w-full" slot="css">
    ```css
    .container {
        --background-image: url('https://s3-us-west-1.amazonaws.com/datyayu-xyz/others/bg.jpg');
        position: relative;
        margin: 0;
        cursor: none;
        overflow: hidden;
        height: 30vh;
        width: 100vw
    }

    .bg {
        width: 100%;
        height: 100%;
        position: absolute;
        top: 0;
        left: 0;
        background-image: var(--background-image);
        background-attachment: fixed;
        background-size: cover;
        background-repeat: no-repeat;
        filter: blur(3px);
        +filter: blur(3px);
    }

    .circle {
        width: 200px;
        height: 200px;
        position: absolute;
        top: calc(50vh - 200px);
        left: calc(50vw - 100px);
        border-radius: 50%;
        border: 3px solid white;
        background-image: var(--background-image);
        background-attachment: fixed;
        background-size: cover;
        background-repeat: no-repeat;
        pointer-events: none;
    }
    ```
</div>


<div class="w-full" slot="js">
     ```js
     const $circle = document.querySelector('#circle');
     const $container = document.querySelector('#container');
     const CIRCLE_RADIUS = 200;

     $container.addEventListener('mousemove', (event) => {
         const x = event.offsetX;
         const y = event.offsetY;

         $circle.style.top = `${y - (CIRCLE_RADIUS/2)}px`;
         $circle.style.left = `${x - (CIRCLE_RADIUS/2)}px`;
     });
     ```
 </div>
